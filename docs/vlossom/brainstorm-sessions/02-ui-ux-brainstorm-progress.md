  xpShown: "hidden" | "secondary" | "details_only";
  badgeShown: "none" | "subtle" | "details_only";

export function resolveProgressionUIState(
  hair: HairHealthSignals,
  prog: ProgressionSignals,
  ctx: UIContext
): ProgressionUIState {
  const xpTier = prog.xpTier ?? computeXpTier(prog.xpTotal);
  const repState = prog.reputationState ?? computeReputationState(prog.reputationScore);

  const primaryState = computePrimaryState(hair);
  const primaryIcon = primaryIconFor(primaryState, xpTier);

  const { xpShown, badgeShown, visibilityLevel } = defaultsForSurface(ctx.surface);

  const secondaryIcons: IconState[] = [];

  // Trust indicators are allowed only on identity-heavy surfaces
  if (["profile_overview", "stylist_dashboard", "salon_dashboard"].includes(ctx.surface)) {
    secondaryIcons.push(trustIconFor(repState));
  }

  // Badge markers: show only if relevant to the surface, and always subtle
  const latestBadge = mostRecentBadge(prog.badges);
  const recentlyEarnedBadge = latestBadge ? isRecent(latestBadge.earnedAt, 7) : false;

  if (latestBadge) {
    const badgeIcon = badgeIconFor(latestBadge.type);
    if (badgeIcon && shouldShowBadgeOnSurface(latestBadge.type, ctx.surface)) {
      secondaryIcons.push(badgeIcon);
    }
  }

  // Enforce max 2 secondary icons
  const clippedSecondary = secondaryIcons.slice(0, 2);

  const accentAllowed = computeAccentAllowed(hair, primaryState, recentlyEarnedBadge);

  const messageTone = toneFor(primaryState, ctx.surface, hair.confidence);

  return {
    primaryState,
    primaryIcon,
    secondaryIcons: clippedSecondary,
    accentAllowed,
    messageTone,
    visibilityLevel,
    xpShown,
    badgeShown,
  };
}

function computeXpTier(xpTotal: number): 1 | 2 | 3 | 4 {
  // Replace thresholds later with tuned economy values
  if (xpTotal < 100) return 1;
  if (xpTotal < 300) return 2;
  if (xpTotal < 700) return 3;
  return 4;
}

function computeReputationState(score: number): "forming" | "stable" | "established" | "fragile" {
  // "fragile" should ideally be derived from reliability signals; keep score-based fallback minimal.
  if (score < 30) return "forming";
  if (score < 70) return "stable";
  return "established";
}

function computePrimaryState(h: HairHealthSignals): ProgressionUIState["primaryState"] {
  if (h.overloadDetected) return "care";
  if (h.restRequired || h.growthPhase === "rest" || h.growthPhase === "recovery") return "rest";
  if (h.transitionWindow) return "transition";
  if (h.growthPhase === "active") return "growth";
  return "stable";
}

function primaryIconFor(state: ProgressionUIState["primaryState"], xpTier: 1 | 2 | 3 | 4): IconState {
  if (state === "care") return "state-needs-care";
  if (state === "rest") return "state-resting";
  if (state === "transition") return "state-transition";
  if (state === "growth") return (`growth-stage-${xpTier}` as IconState);
  return "state-healthy";
}

function trustIconFor(rep: "forming" | "stable" | "established" | "fragile"): IconState {
  switch (rep) {
    case "forming":
      return "trust-forming";
    case "stable":
      return "trust-stable";
    case "established":
      return "trust-established";
    case "fragile":
      return "trust-fragile";
  }
}

function badgeIconFor(type: Badge["type"]): IconState | null {
  switch (type) {
    case "ritual":
      return "ritual-mark";
    case "learning":
      return "learning-mark";
    case "trust":
      return "verified-mark";
    case "milestone":
      return "growth-mark";
    default:
      return null;
  }
}

function shouldShowBadgeOnSurface(type: Badge["type"], surface: UIContext["surface"]): boolean {
  // Keep badges out of high-cognition surfaces like map overlays and calendar grids
  if (["home_map_overlay", "calendar_month"].includes(surface)) return false;

  // Ritual badges make sense in notifications and calendar-day details
  if (type === "ritual") return ["notification_item", "calendar_day", "profile_overview"].includes(surface);

  // Learning badges in profile + onboarding-related notifications
  if (type === "learning") return ["profile_overview", "notification_item"].includes(surface);

  // Trust badges on profile/stylist/salon surfaces
  if (type === "trust") return ["profile_overview", "stylist_dashboard", "salon_dashboard"].includes(surface);

  // Milestones can appear in profile or notifications
  if (type === "milestone") return ["profile_overview", "notification_item"].includes(surface);

  return false;
}

function mostRecentBadge(badges: Badge[]): Badge | null {
  if (!badges.length) return null;
  return [...badges].sort((a, b) => Date.parse(b.earnedAt) - Date.parse(a.earnedAt))[0]!;
}

function isRecent(earnedAtIso: string, days: number): boolean {
  const earnedAt = Date.parse(earnedAtIso);
  const now = Date.now();
  const ms = days * 24 * 60 * 60 * 1000;
  return now - earnedAt <= ms;
}

function computeAccentAllowed(
  hair: HairHealthSignals,
  primaryState: ProgressionUIState["primaryState"],
  recentlyEarnedBadge: boolean
): boolean {
  // Orange is sacred: never for care/rest moments
  if (primaryState === "care" || primaryState === "rest") return false;

  // Transition is allowed (earned change)
  if (hair.transitionWindow) return true;

  // Badge acknowledgment is allowed (subtle)
  if (recentlyEarnedBadge) return true;

  // Growth may use accent, but must stay below ~8% surface area
  if (primaryState === "growth") return true;

  return false;
}

function toneFor(
  state: ProgressionUIState["primaryState"],
  surface: UIContext["surface"],
  confidence: HairHealthSignals["confidence"]
): ProgressionUIState["messageTone"] {
  if (state === "care") return "reassure";
  if (state === "rest") return "reassure";
  if (state === "transition") return "inform";
  if (state === "growth") return confidence === "low" ? "encourage" : "acknowledge";
  return "inform";
}

function defaultsForSurface(surface: UIContext["surface"]): Pick<
  ProgressionUIState,
  "xpShown" | "badgeShown" | "visibilityLevel"
> {
  switch (surface) {
    case "calendar_day":
    case "calendar_month":
    case "home_map_overlay":
      return { xpShown: "hidden", badgeShown: "none", visibilityLevel: "minimal" };
    case "notification_item":
      return { xpShown: "hidden", badgeShown: "subtle", visibilityLevel: "standard" };
    case "wallet_summary":
      return { xpShown: "secondary", badgeShown: "details_only", visibilityLevel: "standard" };
    case "stylist_dashboard":
    case "salon_dashboard":
      return { xpShown: "secondary", badgeShown: "subtle", visibilityLevel: "expanded" };
    case "profile_overview":
    default:
      return { xpShown: "secondary", badgeShown: "subtle", visibilityLevel: "standard" };
  }
}
